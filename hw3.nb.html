read_hpi <- function(hpi_file_name) {
  
  message(hpi_file_name)
  
  hpi <- read_excel(hpi_file_name, 2)  
  
  cols <- colnames(hpi)
  
  cols[1] <- "quarter"
  cols[is.na(cols)] <- "JUNK"
  
  cols <- tolower(cols) %>% str_replace_all("[[:punct:][:space:]]+", "_")
  
  hpi %>% 
    setNames(cols)
}
